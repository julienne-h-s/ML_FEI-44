def preprocess_text(text):
    text = str(text).lower()
    text = re.sub(r"http\S+|www\S+|https\S+", '', text)
    text = re.sub(r'[^a-z\s]', '', text)
    tokens = text.split()
    tokens = [w for w in tokens if w not in stop_words]
    return tokens

train_df['tokens'] = train_df['OriginalTweet'].apply(preprocess_text)
test_df['tokens']  = test_df['OriginalTweet'].apply(preprocess_text)

vector_size = 200
w2v_model = Word2Vec(sentences=train_df['tokens'], vector_size=vector_size, window=5, min_count=2, workers=4, sg=1)

def tweet_vector(tokens, model, vector_size):
    vec = np.zeros(vector_size)
    count = 0
    for word in tokens:
        if word in model.wv:
            vec += model.wv[word]
            count += 1
    if count > 0:
        vec /= count
    return vec

X_train = np.array([tweet_vector(t, w2v_model, vector_size) for t in train_df['tokens']])
X_test  = np.array([tweet_vector(t, w2v_model, vector_size) for t in test_df['tokens']])

le = LabelEncoder()
y_train = le.fit_transform(train_df['Sentiment'])
y_test  = le.transform(test_df['Sentiment'])

# === :       Word2Vec ===
def text_to_vector(text, w2v_model, vector_size):
    words = text.split()
    vecs = [w2v_model.wv[w] for w in words if w in w2v_model.wv]
    if len(vecs) == 0:  #      
        return np.zeros(vector_size)
    return np.mean(vecs, axis=0)

# ===   ===
def predict_new_tweet(text, w2v_model, vector_size, model, le, pca=None):
    vec = text_to_vector(text, w2v_model, vector_size).reshape(1, -1)
    if pca is not None:
        vec = pca.transform(vec)
    pred = model.predict(vec)
    return le.inverse_transform(pred)[0]
